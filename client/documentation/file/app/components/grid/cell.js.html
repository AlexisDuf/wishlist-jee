    app/components/grid/cell.js

    import React, {Component} 	from 'react';
import {Provider}		  	from 'react-redux';
import MainTheme 		  	from '../theme/mainTheme.js';
import ThemeManager  	  	from 'material-ui/lib/styles/theme-manager';
import Paper     		  	from 'material-ui/lib/paper';
import rd3 					from 'react-d3';

/**
 * Cell React class
 * @extends {Component}
 */
export default class Cell extends Component{
	
	/**
	* Class constructor.
	*/
	constructor(){
		super();
	}

	/**
     * @type {object}
	* defining the child context types
     */
	static get childContextTypes(){
        return {
			muiTheme: React.PropTypes.object,
		};
    }

    /**
	* get the child context.
	* @returns {object} wich specify the child context
	*/
	getChildContext() {
		return {
			muiTheme: ThemeManager.getMuiTheme(MainTheme),
		};
	}

	/**
	* specify how to render the component.
	* @returns {html} representing the component
	*/
	render(){
		
		let styles = {
			container:{
				height:300,
				display:"flex",
				boxSizing:"border-box",
				width:"32.3%",
				marginRight:".5%",
				marginLeft:".5%",
				marginTop:"10px",
				marginBottom:"10px",
				float:"left"
			},
			chart:{
				margin:"auto"
			}
		};

		let ChartNode = rd3[this.props.config.type];

		return (
		  <Paper style={styles.container} zDepth={1}>
			<div style={styles.chart}>
		  		<ChartNode data={this.props.data} {...this.props.config}/>
			</div>
		  </Paper>
		);
	}
};
