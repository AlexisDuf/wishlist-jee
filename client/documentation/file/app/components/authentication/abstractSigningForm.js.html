    import React, {Component}	from 'react';
import {connect} 			from 'react-redux';
import { browserHistory } 	from 'react-router';
import Authentication 	    from '../../models/authentication/authentication.js';
import MainTheme 		  	from '../theme/mainTheme.js';
import ThemeManager  	  	from 'material-ui/lib/styles/theme-manager';


/**
 * Unbinded sign in form React class
 * @extends {Component}
 */
export default class AbstractSigningForm extends Component{

	/**
	* Class constructor.
	* @param {object} props - properties passed to component.
	*/
	constructor(props){
		super(props)
		/**
	     * @type {object}
	     */
		this.state = {
			email:'',
			password:'',
			email_error_text:null,
			password_error_text:null,
			loading:false,
			loadingState:null
		};
	}

	/**
     * @type {object}
	* defining the child context types
     */
	static get childContextTypes(){
        return {
			muiTheme: React.PropTypes.object,
		};
    }

    /**
     * @type {object}
	* defining the context types
     */
	static get contextTypes(){
        return {
			muiTheme: React.PropTypes.object,
		};
    }

    /**
	* get the child context.
	* @returns {object} wich specify the child context
	*/
	getChildContext() {
		return {
			muiTheme: ThemeManager.getMuiTheme(MainTheme),
		};
	}

	/**
	* [PRIVATE] check the validity of the email and set the state with current value.
	* @param {string} email - the tested email.
	* @returns {boolean} true if valid false if not.
	*/
	_isValidEmail(email){
		if(!email){return false};
		var valid_email_regex = /^(([^<>()[\]\.,;:\s@\"]+(\.[^<>()[\]\.,;:\s@\"]+)*)|(\".+\"))@(([^<>()[\]\.,;:\s@\"]+\.)+[^<>()[\]\.,;:\s@\"]{2,})$/i;
		return valid_email_regex.test(email);
	}

	/**
	* [PRIVATE] check the validity of the email on change and bind the value to the state email.
	* @param {object} event - html event from input.
	*/
	_emailInputChanged(event){
		var email = event.target.value;
		this.setState({email:email});
    	if(this._isValidEmail(email)){
    		this.setState({email_error_text:null});
    	}
	}

	/**
	* [PRIVATE] check the validity of the email after the blur.
	* @param {object} event - html event from input.
	*/
	_emailInputBlured(event){
		var email = event.target.value;
    	if(!this._isValidEmail(email)){
    		this.setState({email_error_text:"Invalid email"});
    	}
	}
	
	/**
	* [PRIVATE] check the validity of the password on change and bind the value to the state password.
	* @param {object} event - html event from input.
	*/
	_passwordInputChanged(event){
		var password = event.target.value;
		this.setState({password_error_text:null, password:password});
	}

	/**
	* [PRIVATE] check the validity of the email after the blur.
	* @param {object} event - html event from input.
	*/
	_passwordInputBlured(event){
		var password = event.target.value;
    	if(!password || password.length<1){
    		this.setState({password_error_text:"This field is required"});
    	}
	}

	/**
	* [PRIVATE] handle the google auth answer
	* @param {object} googleUser - the google user.
	*/
	_onGoogleSignIn(googleUser) {
		var profile = googleUser.getBasicProfile();
		console.log('ID: ' + profile.getId()); // Do not send to your backend! Use an ID token instead.
		console.log('Name: ' + profile.getName());
		console.log('Image URL: ' + profile.getImageUrl());
		console.log('Email: ' + profile.getEmail());
	}

	/**
	* [PRIVATE] handle the fb auth answer
	* @param {object} fbUser - the facebook user.
	*/
	_onFBSignIn(fbUser) {
		if(!(fbUser.status == "unknown")){
			Authentication.facebookConnect(fbUser);
		}
	}

	/*
	*@overide
	*/
	componentWillReceiveProps(nextProps){
		if(nextProps.connected){
			this.props.history.push('/home');
		}
	}

	/*
	*@overide
	*/
	componentDidMount(){
		gapi.signin2.render('g-signin2', {
			'scope': 'profile email',
			'longtitle': false,
			'theme': 'light',
			'width': 250,
			'height': 48,
			'onsuccess': this._onGoogleSignIn
		});

		if(this.props.connected){
			this.props.history.push('/home');
		}
	}
	
	/**
	* specify how to render the component.
	* @returns {html} representing the component
	*/
	render() {
		return (<div></div>);
	}
};
