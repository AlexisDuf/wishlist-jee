    import React, {Component}				from 'react'
import {connect} 						from 'react-redux'
import AbstractSigningForm				from './abstractSigningForm.js';
import TextField 						from 'material-ui/lib/text-field'
import RaisedButton 					from 'material-ui/lib/raised-button'
import ThemeManager  	  				from 'material-ui/lib/styles/theme-manager'
import LinearProgress 					from 'material-ui/lib/linear-progress'
import authenticationStore 				from '../../stores/authentication/authenticationStore.js'
import Authentication 	    			from '../../models/authentication/authentication.js'
import AuthenticationFormContainer		from './authenticationFormContainer.js'
import MainTheme 		  				from '../theme/mainTheme.js'
import FacebookLogin 					from 'react-facebook-login'


/**
 * Unbinded sign up form React class
 * @extends {AbstractSigningForm}
 */
export class SignUpForm extends AbstractSigningForm{

	/**
	* Class constructor.
	* @param {object} props - properties passed to component.
	*/
	constructor(props){
		super(props);
	}

	/**
	* trigger the register process.
	* @param {object} event - html event from button.
	*/
	submit(event){
		Authentication.register(this.state.email, this.state.password);
	}
	
	/**
	* override
	*/
	render() {
		let styles = {
			container:{
				position:"relative"
			},
			title:{
				color:MainTheme.palette.primary1Color,
				fontSize:"30px",
				margin:0,
				padding:"10px"
			},
			form:{
				width:"90%",
				margin:"auto",
				marginTop:"10px",
				marginBottom:"10px"
			},
			row:{
				marginTop:"10px",
				marginBottom:"10px",
				height:'70px'
			},
			emailField:{
				width:"100%"
			},
			passwordField:{
				width:"100%"
			},
			submitButton:{
				width:"20%",
				margin:"auto",
				display:"block"
			}
		};

		return (
			<div style={styles.container}>
				<p style={styles.title}>Sign up</p>
				<div style={styles.form}>
					<div style={styles.row}>
						<TextField disabled={this.props.connectionStatus} style={styles.emailField} hintText="Email" value={this.state.email}  errorText={this.state.email_error_text} onBlur={this._emailInputBlured.bind(this)} onChange={this._emailInputChanged.bind(this)} />
					</div>
					<div style={styles.row}>
				    	<TextField disabled={this.props.connectionStatus} style={styles.passwordField} hintText="Password" type="password" value={this.state.password}  errorText={this.state.password_error_text} onBlur={this._passwordInputBlured.bind(this)} onChange={this._passwordInputChanged.bind(this)} />
					</div>
					<div style={styles.row}>
				    	<RaisedButton style={styles.submitButton} label="Sign in" disabled={(this.props.connectionStatus || this.state.email_error_text || this.state.password_error_text || !this.state.email.length || !this.state.password.length)} primary={true} onClick={this.submit.bind(this)} />
					</div>
			 	</div>
			 	{(this.props.connectionStatus)?<LinearProgress mode="indeterminate"/>:null}
			</div>
		);
	}
};

/**
* Select the properties needed from the store.
* @param {object} state - containing the all store properties.
* @returns {object} the needed subset.
*/
function selectPropertyFromStore(state) {
	return {
		connected: state.AuthenticationReducer.connected,
		access_token: state.AuthenticationReducer.access_token,
		refresh_token: state.AuthenticationReducer.refresh_token,
		token_type: state.AuthenticationReducer.token_type,
		expires_in: state.AuthenticationReducer.expires_in,
		connectionStatus:state.AuthenticationReducer.connectionStatus,
		connectionStep:state.AuthenticationReducer.connectionStep,
	}
}

/**
* An extended verison of {SignInform} witch is connected to the store
*/
export default connect(selectPropertyFromStore)(SignUpForm);
